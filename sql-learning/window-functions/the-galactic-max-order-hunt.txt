WITH max_amount_per_customer AS (
	SELECT
		customerid,
		orderid,
		orderdate,
		totalamount,
		MAX(totalamount) OVER(PARTITION BY customerid) as maxorderamount
	FROM orders
)

SELECT
	*
FROM max_amount_per_customer as m
ORDER BY customerid, orderdate
