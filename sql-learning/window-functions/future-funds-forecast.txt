# https://www.practicewindowfunctions.com/problem/55/future-funds-forecast

WITH base AS (
	SELECT
		o.customerid,
		t.transactionid,
		o.totalamount,
		LEAD(t.transactionid, 2) OVER(PARTITION BY o.customerid ORDER BY t.transactiondate) as secondnexttransactionid,
		LEAD(o.totalamount, 2) OVER(PARTITION BY o.customerid ORDER BY t.transactiondate) as secondnexttransactionamount
	FROM orders as o
	JOIN transaction as t ON o.orderid = t.orderid
)

SELECT * FROM base
