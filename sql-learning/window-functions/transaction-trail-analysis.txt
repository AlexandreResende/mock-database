# https://www.practicewindowfunctions.com/problem/73/transaction-trail-analysis

WITH base AS (
	SELECT
		c.species,
		o.orderid as transactionid,
		o.totalamount
	FROM customers as c
	JOIN orders as o ON c.customerid = o.customerid
)

SELECT
	*,
	RANK() OVER(PARTITION BY species ORDER BY totalamount DESC) as transactionrank,
	totalamount - LAG(totalamount) OVER(PARTITION BY species ORDER BY totalamount DESC) as differencefromprevious
FROM base
