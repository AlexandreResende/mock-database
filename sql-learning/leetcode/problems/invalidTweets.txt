# https://leetcode.com/problems/invalid-tweets/description/
# My solution

SELECT tweet_id
FROM Tweets t
WHERE CHAR_LENGTH(content) > 15

SELECT tweet_id
FROM Tweets
WHERE content REGEXP '^[A-Za-z0-9 !]{16,}$';

SELECT tweet_id
FROM Tweets
WHERE content NOT REGEXP '^[A-Za-z0-9 !]{1,15}$';

SELECT tweet_id
FROM Tweets
WHERE NOT REGEXP_LIKE(content, '^[A-Za-z0-9 !]{1,15}$');
