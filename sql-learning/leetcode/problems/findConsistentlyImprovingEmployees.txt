# https://leetcode.com/problems/find-consistently-improving-employees/description/

WITH base AS (
    SELECT
        e.employee_id,
        e.name as name,
        p.review_date,
        p.rating,
        ROW_NUMBER() OVER(PARTITION BY e.employee_id ORDER BY p.review_date DESC) as r 
    FROM employees as e
    JOIN performance_reviews as p ON e.employee_id = p.employee_id
), base2 AS (
    SELECT
        employee_id,
        name,
        r,
        IF(r=3, rating, 0) as first_prev,
        IF(r=2, rating, 0) as second_prev,
        IF(r=1, rating, 0) as third_prev
    FROM base
), base3 AS (
    SELECT
        employee_id,
        name,
        SUM(first_prev) as first_prev,
        SUM(second_prev) as second_prev,
        SUM(third_prev) as third_prev
    FROM base2
    WHERE r <= 3
    GROUP BY employee_id
), base4 AS (
    SELECT
        *,
        third_prev - first_prev as improvement_score
    FROM base3
    WHERE first_prev < second_prev AND second_prev < third_prev AND
    second_prev <> 0 AND first_prev <> 0 AND third_prev <> 0
)

SELECT 
    employee_id,
    name,
    improvement_score 
FROM base4
ORDER BY improvement_score DESC, name

WITH RankedReviews AS (
    SELECT pr.employee_id, pr.review_date, e.name, pr.rating,
           ROW_NUMBER() OVER (PARTITION BY pr.employee_id ORDER BY pr.review_date DESC) AS rn
    FROM performance_reviews pr
    JOIN employees e
        ON e.employee_id = pr.employee_id
),
Last3 AS (
    SELECT *
    FROM RankedReviews
    WHERE rn <= 3
),
Increase AS (
    SELECT *,
           LAG(rating, 1) OVER (PARTITION BY employee_id ORDER BY review_date) AS prev,
           LAG(rating, 2) OVER (PARTITION BY employee_id ORDER BY review_date) AS prev2
    FROM Last3
)

SELECT employee_id, name, rating - prev2 AS improvement_score
FROM Increase
WHERE prev2 IS NOT NULL
  AND rating > prev
  AND prev > prev2
ORDER BY improvement_score DESC, name;
