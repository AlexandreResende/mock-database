# https://leetcode.com/problems/trips-and-users/description/
# My solution

SELECT request_at as Day, ROUND(SUM(IF(t.status = 'cancelled_by_driver' OR t.status = 'cancelled_by_client', 1, 0))/COUNT(*), 2) as "Cancellation Rate"
FROM Trips t
JOIN Users u1
ON t.client_id = u1.user_id
JOIN User u2
ON t.driver_id = u2.user_id
WHERE u1.banned <> "Yes"
AND u2.banned <> "Yes"
AND t.request_at BETWEEN "2013-10-01" AND "2013-10-03"
GROUP BY request_at;
