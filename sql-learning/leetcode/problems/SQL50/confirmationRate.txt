# https://leetcode.com/problems/confirmation-rate/description/?envType=study-plan-v2&envId=top-sql-50
# My solution

SELECT 
    s.user_id,
    ROUND(
        (SELECT COUNT(*) FROM Confirmations WHERE user_id = s.user_id AND action = 'confirmed') / COUNT(*), 2
    ) as confirmation_rate
FROM Signups s
LEFT JOIN Confirmations c
ON s.user_id = c.user_id
GROUP BY s.user_id;

# Other solution

SELECT
  s.user_id,
  ROUND(
    AVG(
        IF(c.action = 'confirmed', 1.00, 0)
    ), 
    2) confirmation_rate
FROM Signups s
LEFT JOIN Confirmations c
ON s.user_id = c.user_id
GROUP BY s.user_id
