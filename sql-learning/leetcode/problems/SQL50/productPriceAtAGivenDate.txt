# https://leetcode.com/problems/product-price-at-a-given-date/description/?envType=study-plan-v2&envId=top-sql-50
# My solution

WITH rank_prices_on_date AS (
    SELECT 
        product_id, 
        new_price, 
        change_date, 
        ROW_NUMBER() OVER(PARTITION BY product_id ORDER BY change_date DESC) as ranking
    FROM Products
    WHERE change_date <= '2019-08-16'
)

(SELECT 
    product_id,
    new_price as price
FROM rank_prices_on_date
WHERE ranking = 1)
UNION
(SELECT
    product_id,
    10 as price
FROM Products
WHERE product_id NOT IN (SELECT DISTINCT product_id FROM rank_prices_on_date)
)
