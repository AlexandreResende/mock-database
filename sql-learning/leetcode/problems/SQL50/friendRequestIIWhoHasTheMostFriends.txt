# https://leetcode.com/problems/friend-requests-ii-who-has-the-most-friends/description/?envType=study-plan-v2&envId=top-sql-50
# My solution

WITH Result_Counter AS(
    SELECT
        requester_id as id
    FROM RequestAccepted
    UNION ALL
    SELECT
        accepter_id as id
    FROM RequestAccepted
)

SELECT
    id,
    COUNT(*) num
FROM Result_Counter
GROUP BY id
ORDER BY num DESC
LIMIT 1;
