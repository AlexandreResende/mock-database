# https://leetcode.com/problems/monthly-transactions-i/?envType=study-plan-v2&envId=top-sql-50
# My solution

SELECT
    CONCAT(
        YEAR(trans_date), '-', IF(MONTH(trans_date) < 10, CONCAT('0', MONTH(trans_date)), MONTH(trans_date))
    ) as month,
    country,
    COUNT(*) as trans_count,
    COUNT(IF(state = 'approved', amount, null)) as approved_count,
    SUM(amount) as trans_total_amount,
    SUM(IF(state = 'approved', amount, 0)) as approved_total_amount
FROM Transactions
GROUP BY month, country;

# Other solution

SELECT 
    LEFT(trans_date, 7) AS month,
    country, 
    COUNT(id) AS trans_count,
    SUM(state = 'approved') AS approved_count,
    SUM(amount) AS trans_total_amount,
    SUM((state = 'approved') * amount) AS approved_total_amount
FROM 
    Transactions
GROUP BY 
    month, country;
