# https://leetcode.com/problems/find-product-recommendation-pairs/description/

WITH base AS (
    SELECT
        pp.user_id as user_id,
        pp.product_id as product1_id,
        p.category as product1_category,
        pp1.product_id as product2_id,
        p1.category as product2_category
    FROM ProductPurchases as pp
    JOIN ProductInfo as p ON pp.product_id = p.product_id
    JOIN ProductPurchases as pp1 ON pp1.product_id > pp.product_id AND pp1.user_id = pp.user_id
    JOIN ProductInfo as p1 ON pp1.product_id = p1.product_id
)

SELECT
    product1_id,
    product2_id,
    product1_category,
    product2_category,
    COUNT(user_id) as customer_count
FROM base
GROUP BY product1_id, product2_id
HAVING COUNT(user_id) >= 3
ORDER BY customer_count DESC, product1_id, product2_id
