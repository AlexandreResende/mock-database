# https://leetcode.com/problems/analyze-subscription-conversion/description/

WITH base AS (
    SELECT
        user_id,
        activity_type,
        activity_date,
        CASE
            WHEN activity_type = 'free_trial' THEN activity_duration
            ELSE 0
        END as trial_duration,
        CASE
            WHEN activity_type = 'paid' THEN activity_duration
            ELSE 0
        END as paid_duration
    FROM UserActivity
    WHERE user_id IN (SELECT DISTINCT user_id FROM UserActivity WHERE activity_type = 'paid')
), base2 AS (
    SELECT
        user_id,
        ROUND(SUM(trial_duration)/COUNT(*), 2) as trial_avg_duration,
        ROUND(SUM(paid_duration)/COUNT(*), 2) as paid_avg_duration
    FROM base
    GROUP BY user_id, activity_type
)

SELECT 
    user_id,
    SUM(trial_avg_duration) as trial_avg_duration,
    SUM(paid_avg_duration) as paid_avg_duration
FROM base2 
GROUP BY user_id;
