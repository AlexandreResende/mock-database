# my worst solution EVER
SELECT 
    emp.employee_id, 
    emp.name, 
    (
        SELECT COUNT(*) FROM Employees WHERE reports_to = emp.employee_id
    ) as reports_count, 
    (
        SELECT ROUND(AVG(age),0) FROM Employees WHERE reports_to = emp.employee_id
    ) as average_age
FROM Employees emp
WHERE (
    SELECT COUNT(*) FROM Employees WHERE reports_to = emp.employee_id
) > 0
ORDER BY emp.employee_id;

select 
  e1.employee_id,
  e1.name,
  count(e2.reports_to) reports_count,
  round(avg(e2.age*1.00),0) average_age
from employees e1 
inner join employees e2 
on e2.reports_to=e1.employee_id
group by e1.employee_id,e1.name
order by e1.employee_id
