# https://leetcode.com/problems/biggest-single-number/description/
# My solution

SELECT num
FROM MyNumbers
GROUP BY num
HAVING COUNT(*) = 1
UNION SELECT null
ORDER BY num DESC 
LIMIT 1;

# Other solution

SELECT MAX(num) AS num
FROM (
    SELECT num
    FROM MyNumbers
    GROUP BY num
    HAVING COUNT(num) = 1
) AS unique_numbers;
