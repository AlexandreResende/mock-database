# https://leetcode.com/problems/students-and-examinations/description/
# My solution

SELECT 
  st.student_id,
  st.student_name,
  s.subject_name,
  IFNULL(COUNT(ex.student_id), 0) as attended_exams
FROM Students st
CROSS JOIN Subjects s
LEFT JOIN Examinations ex
ON (ex.student_id = st.student_id AND s.subject_name = ex.subject_name) OR ex.student_id IS NULL
GROUP BY st.student_id, s.subject_name
ORDER BY st.student_id, s.subject_name;
