# https://leetcode.com/problems/movie-rating/description/?envType=study-plan-v2&envId=top-sql-50
# My solution

(SELECT
    u.name as results
FROM Users u
JOIN MovieRating m
ON u.user_id = m.user_id
GROUP BY m.user_id
ORDER BY COUNT(m.rating) DESC, u.name
LIMIT 1)
UNION ALL
(SELECT
    m.title as results
FROM Movies m
JOIN MovieRating mr
ON m.movie_id = mr.movie_id
WHERE mr.created_at >= '2020-02-01' AND mr.created_at <= '2020-02-29'
GROUP BY m.movie_id
ORDER BY AVG(mr.rating) DESC, m.title
LIMIT 1);
