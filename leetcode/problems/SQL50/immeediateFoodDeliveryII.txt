# My solution

SELECT 
  (
    ROUND(
      (
        SELECT 
          COUNT(*) 
        FROM 
          Delivery d 
        WHERE 
          order_date IN (
            SELECT 
              MIN(order_date) 
            FROM 
              Delivery 
            WHERE 
              customer_id = d.customer_id
          ) 
          AND order_date = customer_pref_delivery_date
      ) * 100 / COUNT(DISTINCT customer_id), 
      2
    )
  ) as immediate_percentage 
FROM 
  Delivery;

## Getting the first order from each customer
SELECT delivery_id 
FROM Delivery d 
WHERE order_date IN (
    SELECT MIN(order_date) FROM Delivery WHERE customer_id = d.customer_id
) AND order_date = customer_pref_delivery_date;

Select round(avg(order_date = customer_pref_delivery_date)*100, 2) as immediate_percentage
from Delivery
where (customer_id, order_date) in (
  Select customer_id, min(order_date) 
  from Delivery
  group by customer_id
);
